# 2020/09-19/-真值(Truthy)與假值(Falsy)

if (1){
    console.log('真值');//真值
}
else{
    console.log('假值');
}

if (0){
    console.log('真值');
}
else{
    console.log('假值');//假值
}


//空陣列
if ([]){
    console.log('真值');//真值
}
else{
    console.log('假值');
}
//空物件
if ({}){
    console.log('真值');//真值
}
else{
    console.log('假值');
}

//null
if (null){
    console.log('真值');
}
else{
    console.log('假值');//假值
}
//undefined
if (undefined){
    console.log('真值');
}
else{
    console.log('假值');//假值
}
